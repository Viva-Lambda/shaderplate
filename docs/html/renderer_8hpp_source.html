8 glm::vec3 ‪getTangent(glm::vec2 deltaUV2, glm::vec2 deltaUV1, glm::vec3 edge1,
9  glm::vec3 edge2) {
10  // Compute tangent
11  GLfloat f = 1.0f / (deltaUV1.x * deltaUV2.y - deltaUV2.x * deltaUV1.y);
12  glm::vec3 tangent;
13  tangent.x = f * (deltaUV2.y * edge1.x - deltaUV1.y * edge2.x);
14  tangent.y = f * (deltaUV2.y * edge1.y - deltaUV1.y * edge2.y);
15  tangent.z = f * (deltaUV2.y * edge1.z - deltaUV1.y * edge2.z);
16  tangent = glm::normalize(tangent);
17  return tangent;
18 }

427 void ‪renderSphere() {
428  unsigned int sphereVAO;
429  glGenVertexArrays(1, &sphereVAO);
430  unsigned int indexCount;
431 
432  unsigned int vbo, ebo;
433  glGenBuffers(1, &vbo);
434  glGenBuffers(1, &ebo);
435 
436  std::vector<glm::vec3> positions;
437  std::vector<glm::vec2> uv;
438  std::vector<glm::vec3> normals;
439  std::vector<unsigned int> indices;
440 
441  const unsigned int X_SEGMENTS = 64;
442  const unsigned int Y_SEGMENTS = 64;
443  const float PI = 3.14159265359;
444  for (unsigned int y = 0; y <= Y_SEGMENTS; ++y) {
445  for (unsigned int x = 0; x <= X_SEGMENTS; ++x) {
446  float xSegment = (float)x / (float)X_SEGMENTS;
447  float ySegment = (float)y / (float)Y_SEGMENTS;
448  float xPos = std::cos(xSegment * 2.0f * PI) * std::sin(ySegment * PI);
449  float yPos = std::cos(ySegment * PI);
450  float zPos = std::sin(xSegment * 2.0f * PI) * std::sin(ySegment * PI);
451 
452  positions.push_back(glm::vec3(xPos, yPos, zPos));
453  uv.push_back(glm::vec2(xSegment, ySegment));
454  normals.push_back(glm::vec3(xPos, yPos, zPos));
455  }
456  }
457 
458  bool oddRow = false;
459  for (unsigned int y = 0; y < Y_SEGMENTS; ++y) {
460  if (!oddRow) // even rows: y == 0, y == 2; and so on
461  {
462  for (unsigned int x = 0; x <= X_SEGMENTS; ++x) {
463  indices.push_back(y * (X_SEGMENTS + 1) + x);
464  indices.push_back((y + 1) * (X_SEGMENTS + 1) + x);
465  }
466  } else {
467  for (int x = X_SEGMENTS; x >= 0; --x) {
468  indices.push_back((y + 1) * (X_SEGMENTS + 1) + x);
469  indices.push_back(y * (X_SEGMENTS + 1) + x);
470  }
471  }
472  oddRow = !oddRow;
473  }
474  indexCount = indices.size();
475 
476  std::vector<float> data;
477  for (unsigned int i = 0; i < positions.size(); ++i) {
478  data.push_back(positions[i].x);
479  data.push_back(positions[i].y);
480  data.push_back(positions[i].z);
481  if (uv.size() > 0) {
482  data.push_back(uv[i].x);
483  data.push_back(uv[i].y);
484  }
485  if (normals.size() > 0) {
486  data.push_back(normals[i].x);
487  data.push_back(normals[i].y);
488  data.push_back(normals[i].z);
489  }
490  }
491  glBindVertexArray(sphereVAO);
492  glBindBuffer(GL_ARRAY_BUFFER, vbo);
493  glBufferData(GL_ARRAY_BUFFER, data.size() * sizeof(float), &data[0],
494  GL_STATIC_DRAW);
495  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ebo);
496  glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(unsigned int),
497  &indices[0], GL_STATIC_DRAW);
498  float stride = (3 + 2 + 3) * sizeof(float);

500  glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, stride, (void *)0);
