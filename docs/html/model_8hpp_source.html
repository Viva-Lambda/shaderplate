329 unsigned int ‪getTextureFromFile(const char *path, bool gamma) {
330  GLuint tex;
331  glGenTextures(1, &tex);
332  //
333  int width, height, nbChannels;
334  unsigned char *data = stbi_load(path, &width, &height, &nbChannels, 0);
335  if (data) {
336  GLenum format;
337  if (nbChannels == 1) {
338  format = GL_RED;
339  } else if (nbChannels == 3) {
340  format = GL_RGB;
341  } else if (nbChannels == 4) {
342  format = GL_RGBA;
343  }
344  glBindTexture(GL_TEXTURE_2D, tex);
345  glTexImage2D(GL_TEXTURE_2D, 0, format, width, height, 0, format,
346  GL_UNSIGNED_BYTE, data);
347  glGenerateMipmap(GL_TEXTURE_2D);
348 
349  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
350  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
351 
352  // use nearest neighbor interpolation when zooming out
353  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
354 
355  // use nearest neighbor interpolation when zooming out
356  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
357 
358  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER,
359  GL_LINEAR_MIPMAP_LINEAR);
360 
361  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
362 
363  } else {
364  std::cout << "Failed to load texture" << std::endl;
365  }
366  stbi_image_free(data);
367  return tex;
368 }
